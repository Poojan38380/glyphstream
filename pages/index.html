Quick Start

// Each component is a single HTML page + TypeScript module // Import and use directly in your project: import { AsciiFlowField } from './src/components/ascii-flow-field' const art = new AsciiFlowField('#container', { cols: 70, rows: 35, particleCount: 600, noiseScale: 0.005, colorMode: 'monochrome', seed: 42, }) art.start() art.setPreset('turbulent') art.regenerate(123)
